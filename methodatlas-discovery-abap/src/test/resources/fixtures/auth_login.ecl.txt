ATTRIBUTES
  TITLE 'Authentication Tests'
  VERSION '2'
  RISK HIGH
  TRANSPORT NONE
END_ATTRIBUTES

FUNCTION Z_AUTH_LOGIN_VALID
  TITLE 'Test valid login succeeds'
  PARAMS
  ENDPARAMS
  DO
    CHECK 'Z_AUTH_CHECK' TCODE USING_VARIANT 'LOGIN_VALID'
    MARK 'Login valid' STATUS 'OK'
  DONE

FUNCTION Z_AUTH_LOGIN_INVALID
  TITLE 'Test invalid password is rejected'
  PARAMS
  ENDPARAMS
  DO
    CHECK 'Z_AUTH_CHECK' TCODE USING_VARIANT 'LOGIN_INVALID'
    MARK 'Login invalid' STATUS 'OK'
  DONE

FUNCTION Z_AUTH_LOGOUT
  TITLE 'Test logout clears session'
  PARAMS
  ENDPARAMS
  DO
    CHECK 'Z_SESSION_CHECK' TCODE USING_VARIANT 'LOGOUT'
    MARK 'Logout' STATUS 'OK'
  DONE
